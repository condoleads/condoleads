L  40: /**
L  41:  * Sends a tenant-owned email via the tenant's own Resend credentials.
L  42:  * Throws TenantEmailNotConfigured if the tenant is not set up.
L  43:  * Throws TenantEmailFailed if Resend rejects the send.
L  44:  *
L  45:  * Caller pattern:
L  46:  *   try { await sendTenantEmail({ tenantId, to, subject, html }) }
L  47:  *   catch (e) {
L  48:  *     if (e instanceof TenantEmailNotConfigured) { ... soft-fail, lead still captured ... }
L  49:  *     if (e instanceof TenantEmailFailed) { ... return 502 ... }
L  50:  *     throw e
L  51:  *   }
L  52:  */
L  53: export async function sendTenantEmail(
L  54:   params: SendTenantEmailParams
L  55: ): Promise<SendTenantEmailResult> {
L  56:   const supabase = createClient()
L  57: 
L  58:   const { data: tenant, error: dbError } = await supabase
L  59:     .from('tenants')
L  60:     .select('resend_api_key, email_from_domain, send_from, resend_verification_status, brand_name')
L  61:     .eq('id', params.tenantId)
L  62:     .maybeSingle()
L  63: 
L  64:   if (dbError) {
L  65:     throw new TenantEmailNotConfigured(params.tenantId, [`db error: ${dbError.message}`])
L  66:   }
L  67:   if (!tenant) {
L  68:     throw new TenantEmailNotConfigured(params.tenantId, ['tenant not found'])
L  69:   }
L  70: 
L  71:   const missing: string[] = []
L  72:   if (!tenant.resend_api_key) missing.push('resend_api_key')
L  73:   if (!tenant.email_from_domain) missing.push('email_from_domain')
L  74:   if (!tenant.send_from) missing.push('send_from')
L  75:   if (tenant.resend_verification_status !== 'verified') {
L  76:     missing.push(`domain not verified (status=${tenant.resend_verification_status ?? 'null'})`)
L  77:   }
L  78:   if (missing.length > 0) {
L  79:     throw new TenantEmailNotConfigured(params.tenantId, missing)
L  80:   }
L  81: 
L  82:   const resend = new Resend(tenant.resend_api_key!)
L  83: 
L  84:   const { data, error: sendError } = await resend.emails.send({
L  85:     from: tenant.send_from!,
L  86:     to: params.to,
L  87:     cc: params.cc,
L  88:     bcc: params.bcc,
L  89:     replyTo: params.replyTo,
L  90:     subject: params.subject,
L  91:     html: params.html,
L  92:     text: params.text,
L  93:   })
L  94: 
L  95:   if (sendError) {
L  96:     throw new TenantEmailFailed(params.tenantId, sendError)
L  97:   }
L  98:   if (!data?.id) {
L  99:     throw new TenantEmailFailed(params.tenantId, { reason: 'no id returned from Resend' })
L 100:   }
L 101: 
L 102:   return { id: data.id, from: tenant.send_from! }
L 103: }

L 137: export function can(
L 138:   actor: ActorPermissionContext,
L 139:   action: PermAction,
L 140:   target: TargetSpec
L 141: ): CanResult {
L 142:   // Auth gate — must have at least an agent row OR a platform_admins row.
L 143:   if (!actor.agentId && !actor.platformTier) {
L 144:     return UNAUTHORIZED
L 145:   }
L 146: 
L 147:   // Self-protection: role/parent changes on the actor's own agent row are
L 148:   // always blocked, regardless of authority source. Applies before delegation
L 149:   // overlay so it cannot be bypassed via delegation.
L 150:   if (
L 151:     target.kind === 'agent' &&
L 152:     actor.agentId !== null &&
L 153:     target.agentId === actor.agentId &&
L 154:     (action === 'agent.promote' ||
L 155:       action === 'agent.demote' ||
L 156:       action === 'agent.reassignParent')
L 157:   ) {
L 158:     return FORBIDDEN_SELF
L 159:   }
L 160: 
L 161:   // Defence-in-depth: 'delegation.grant' requires the actor to hold their own
L 162:   // authority. Pure delegates (no own role/tier) cannot grant. DB layer also
L 163:   // enforces this via the $sos$ trigger from R2.2.
L 164:   if (action === 'delegation.grant' && !actor.roleDb && !actor.platformTier) {
L 165:     return FORBIDDEN_SOS
L 166:   }
L 167: 
L 168:   // Build effective principal list: actor's own authority + active delegators'
L 169:   // authority. For 'delegation.grant', exclude delegators (no SOS).
L 170:   const principals: EffectivePrincipal[] = [
L 171:     {
L 172:       agentId: actor.agentId,
L 173:       tenantId: actor.tenantId,
L 174:       roleDb: actor.roleDb,
L 175:       platformTier: actor.platformTier,
L 176:       managedAgentIds: actor.managedAgentIds,
L 177:       isDelegate: false,
L 178:     },
L 179:     ...actor.activeDelegators.map<EffectivePrincipal>((d) => ({
L 180:       agentId: d.delegatorId,
L 181:       tenantId: d.delegatorTenantId,
L 182:       roleDb: d.delegatorRoleDb,
L 183:       platformTier: null,
L 184:       managedAgentIds: d.delegatorManagedAgentIds,
L 185:       isDelegate: true,
L 186:     })),
L 187:   ]
L 188: 
L 189:   const candidates = action === 'delegation.grant'
L 190:     ? principals.filter((p) => !p.isDelegate)
L 191:     : principals
L 192: 
L 193:   // First permitting principal wins. If none permit, return last denial.
L 194:   let lastDenial: CanResult = FORBIDDEN_SCOPE
L 195:   for (const p of candidates) {
L 196:     const result = evaluateCell(p, action, target)
L 197:     if (result.ok) return result
L 198:     lastDenial = result
L 199:   }
L 200:   return lastDenial
L 201: }

L 216: function evaluateCell(
L 217:   p: EffectivePrincipal,
L 218:   action: PermAction,
L 219:   target: TargetSpec
L 220: ): CanResult {
L 221:   // Tier 6: Admin Platform — anything goes (self-protection already checked upstream).
L 222:   if (p.platformTier === 'admin') {
L 223:     return OK
L 224:   }
L 225: 
L 226:   // Tier 5: Manager Platform — full authority across overseen tenants.
L 227:   // Cannot platform.write (no peer-promotion; no touching Admin Platform).
L 228:   // All other actions: OK across all tenants, including promote/demote of
L 229:   // tenant_admin which is Mgr Platform+ exclusive per locked spec.
L 230:   if (p.platformTier === 'manager') {
L 231:     if (action === 'platform.write') return FORBIDDEN_PLATFORM
L 232:     return OK
L 233:   }
L 234: 
L 235:   // No platform tier → tenant-scoped principal.
L 236:   if (action === 'platform.read' || action === 'platform.write') {
L 237:     return FORBIDDEN_PLATFORM
L 238:   }
L 239: 
L 240:   return evaluateTenantScoped(p, action, target, /*platformOverride=*/ false)
L 241: }

L 243: function evaluateTenantScoped(
L 244:   p: EffectivePrincipal,
L 245:   action: PermAction,
L 246:   target: TargetSpec,
L 247:   platformOverride: boolean
L 248: ): CanResult {

L 291: function evaluateAgentTarget(
L 292:   p: EffectivePrincipal,
L 293:   action: PermAction,
L 294:   target: Extract<TargetSpec, { kind: 'agent' }>
L 295: ): CanResult {
L 296:   switch (action) {
L 297:     case 'agent.read':
L 298:       // Any tenant-resident principal reads agents in their tenant.
L 299:       // Cross-tenant already blocked upstream.
L 300:       return OK
L 301: 
L 302:     case 'agent.write':
L 303:       // B8: tier 1 (agent) cannot write.
L 304:       if (p.roleDb === 'agent') return FORBIDDEN_ROLE
L 305:       // B9: tier 2 (manager) writes within scope (self + direct children).
L 306:       // Tier 3 (area_manager) writes within full subtree (managedAgentIds
L 307:       // populated transitively for area_managers by R3.2).
L 308:       if (p.roleDb === 'manager' || p.roleDb === 'area_manager') {
L 309:         if (isInManagerScope(p, target)) return OK
L 310:         return FORBIDDEN_SCOPE
L 311:       }
L 312:       // Tier 4 (tenant admin): any agent in tenant.
L 313:       if (isTenantAdminTier(p.roleDb)) return OK
L 314:       return FORBIDDEN_ROLE
L 315: 
L 316:     case 'agent.adminMutate':
L 317:       // B10: tier 4+ only.
L 318:       if (isTenantAdminTier(p.roleDb)) return OK
L 319:       return FORBIDDEN_ROLE
L 320: 
L 321:     case 'agent.promote':
L 322:     case 'agent.demote':
L 323:     case 'agent.reassignParent':
L 324:       return evaluateRoleChange(p, target)
L 325: 
L 326:     default:
L 327:       return FORBIDDEN_ROLE
L 328:   }
L 329: }

L 331: function isInManagerScope(
L 332:   p: EffectivePrincipal,
L 333:   target: Extract<TargetSpec, { kind: 'agent' }>
L 334: ): boolean {
L 335:   if (p.agentId === null) return false
L 336:   return (
L 337:     target.agentId === p.agentId ||
L 338:     target.parentId === p.agentId ||
L 339:     p.managedAgentIds.includes(target.agentId)
L 340:   )
L 341: }

L 343: // Role change matrix (locked spec):
L 344: //   Admin Platform   → any direction (handled upstream as OK).
L 345: //   Manager Platform → a ↔ m ↔ am ↔ ta within overseen tenants (handled via
L 346: //                       virtual tenant_admin; can also act on tenant_admin
L 347: //                       targets, which tenant_admin cannot).
L 348: //   Tenant Admin     → a ↔ m ↔ am within tenant. Cannot touch tenant_admin
L 349: //                       targets (Mgr Platform+ only).
L 350: //   Area Manager     → a ↔ m within own subtree.
L 351: //   Manager / Agent  → none.
L 352: //
L 353: // NOTE: This evaluates whether the actor has authority to act on this TARGET.
L 354: // It does not validate the resulting NEW role — R4 transition state machine
L 355: // owns "new role within actor's authority" + cardinality invariants.
L 356: function evaluateRoleChange(
L 357:   p: EffectivePrincipal,
L 358:   target: Extract<TargetSpec, { kind: 'agent' }>
L 359: ): CanResult {
L 360:   // Admin/Manager Platform handled in evaluateCell upstream.
L 361: 
L 362:   if (isTenantAdminTier(p.roleDb)) {
L 363:     // Tenant Admin: cannot act on tenant_admin/admin targets.
L 364:     if (isTenantAdminTier(target.roleDb)) return FORBIDDEN_ROLE
L 365:     return OK
L 366:   }
L 367: 
L 368:   if (p.roleDb === 'area_manager') {
L 369:     if (target.roleDb !== 'agent' && target.roleDb !== 'manager') {
L 370:       return FORBIDDEN_ROLE
L 371:     }
L 372:     if (!isInManagerScope(p, target)) return FORBIDDEN_SCOPE
L 373:     return OK
L 374:   }
L 375: 
L 376:   return FORBIDDEN_ROLE
L 377: }

L 390:   }

L 401:   if (p.roleDb === 'manager' || p.roleDb === 'area_manager') {

L 403:     if (p.agentId !== null && target.agentId === p.agentId) return OK
L 404:     if (p.managedAgentIds.includes(target.agentId)) return OK
L 405:     return FORBIDDEN_SCOPE
L 406:   }
L 407: 
L 408:   // Tenant Admin: any lead in tenant.
L 409:   if (isTenantAdminTier(p.roleDb)) return OK
L 410: 
L 411:   return FORBIDDEN_ROLE
L 412: }

L 416: // ─────────────────────────────────────────────────────────────────────────────
L 417: 
L 418: function evaluateTenantTarget(p: EffectivePrincipal, action: PermAction): CanResult {
L 419:   if (action === 'tenant.read') {
L 420:     // Any tenant-resident principal reads own tenant config.
L 421:     // Cross-tenant already blocked upstream.
