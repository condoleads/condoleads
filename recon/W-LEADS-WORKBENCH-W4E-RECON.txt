L  16: // is promoted to TO so the email is never address-less.

L 235:     .select('id, email, is_active, tier')

L 242:   const adminPlatformEmails: string[] = []
L 243:   for (const row of (adminPlatformRows || []) as Array<{ id: string; email: string | null; is_active: boolean; tier: string }>) {
L 244:     if (row.email) {
L 245:       adminPlatformEmails.push(row.email)
L 246:       resolved.admin_platforms.push(row.email)
L 247:     }
L 248:   }
L 249:   if (adminPlatformEmails.length === 0) {

L 252: 
L 253:   // ─── Assemble TO / CC / BCC ──────────────────────────────────────────────
L 254:   const to: string[] = []

L 256:   const bcc: string[] = []

L 258:   // Layer 1 → TO. Fallback: if no agent, promote Admin Platform to TO (single recipient

L 260:   if (agentEmail) {

L 262:   } else {
L 263:     // No agent assigned — surface Admin Platform as TO so the message is delivered.
L 264:     // Admin Platform is also still in BCC below; routes/inboxes can dedupe.
L 265:     to.push(adminPlatformEmails[0])
L 266:   }
L 267: 
L 268:   // Layer 2 → CC
L 269:   if (managerEmail) cc.push(managerEmail)
L 270: 
L 271:   // Layer 3 → BCC
L 272:   if (areaManagerEmail) bcc.push(areaManagerEmail)
L 273:   // Layer 4 → BCC
L 274:   if (tenantAdminEmail) bcc.push(tenantAdminEmail)

L 276:   for (const emails of delegateEmailsByDelegator.values()) {
L 277:     for (const e of emails) bcc.push(e)
L 278:   }
L 279:   // Layer 5 → BCC
L 280:   for (const e of managerPlatformEmails) bcc.push(e)
L 281:   // Layer 6 → BCC (unconditional)
L 282:   for (const e of adminPlatformEmails) bcc.push(e)
L 283: 
L 284:   // De-dupe each list (an email shouldn't appear twice in the same field)
L 285:   return {
L 286:     to: dedupe(to),
L 287:     cc: dedupe(cc),
L 288:     bcc: dedupe(bcc),
L 289:     resolved,
L 290:   }
L 291: }

L 107: function resolveLayer(
L 108:   email: string,
L 109:   resolved: LeadEmailRecipients['resolved'],
L 110:   envelopePosition: EmailEnvelopePosition
L 111: ): EmailRecipientLayer {
L 112:   if (resolved.agent === email) return 'agent'
L 113:   if (resolved.manager === email) return 'manager'
L 114:   if (resolved.area_manager === email) return 'area_manager'
L 115:   if (resolved.tenant_admin === email) return 'tenant_admin'
L 116:   if (resolved.manager_platforms.includes(email)) return 'platform_manager'
L 117:   if (resolved.admin_platforms.includes(email)) return 'platform_admin'
L 118: 
L 119:   const isDelegate =
L 120:     resolved.agent_delegates.includes(email) ||
L 121:     resolved.manager_delegates.includes(email) ||
L 122:     resolved.area_manager_delegates.includes(email) ||
L 123:     resolved.tenant_admin_delegates.includes(email)
L 124: 
L 125:   if (!isDelegate) {
L 126:     console.warn('[T3 logEmailRecipients] email not classified in resolved chain — recording as tenant_overlay:', {
L 127:       email,
L 128:       envelopePosition,
L 129:     })
L 130:   }
L 131:
